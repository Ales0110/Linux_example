
#Путь к актуальному месту пользователя:
pwd

#Создать папку:
mkdir Name

#Создать 2 папки:
mkdir dir-1 dir-2

#Зайти в папку:
cd Name

#Зайти в любоую папку:
cd dir-1

#Перейти на папку выше:
cd ..

#Создать 2 файла :
touch file-1.txt  file-1.json 

#Вывести список содержимого папки:
 ls -l

#Открыть любой txt файл:
vim file-1.txt

#Написать туда что-нибудь, любой текст:
i

#Сохранить и выйти:
esc :wq

#Переместить 2 файла в любую другую папку:
mv ./file-1.txt file-1.json ./dir1-1  
mv ./{file-1.txt,file-1.json} ./dir1-1
mv ./file-{1.txt,file-1.json} ./dir1-1

#Скопировать 2 файла в любую другую папку:
cp ./file-1.txt file-1.json ./dir1-1

#Найти файл по имени:
find file-1.txt - ищет в текущей папке
find dir1-1/file-1.txt - ищет во вложенной папке dir1-1

#Просмотреть содержимое в реальном времени:
ls|grep fl2 - вывести в терминал все файлы с fl2 в имени
grep -w 'Hello' file-1.txt - позволяет искать по слову целиком
tail -f Name/dir-1/dir1-1/file-1.txt - выводит содержимое файла в режиме рельного времени после сохранения.
echo "Hello" >> file-1.txt - записывает данные (Hello) в файл без его открытия.

#Вывести несколько первых строк из текстового файла:
head -n2 file-1.txt 

#Просмотреть содержимое длинного файла:
less file-1.txt - просмотр содержимого файла в отдельном черном окне
ESC : wq - выход из просмотра

#Вывести дату и время:
date
